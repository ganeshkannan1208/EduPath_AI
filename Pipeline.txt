import json
from pathlib import Path

BASE_DIR = Path(__file__).resolve().parents[1]
GRAPH_DIR = BASE_DIR / "graph"
EXPORT_DIR = GRAPH_DIR / "exports"

EXPORT_DIR.mkdir(parents=True, exist_ok=True)

nodes = json.load(open(GRAPH_DIR / "edu_nodes.json", encoding="utf-8"))
edges = json.load(open(GRAPH_DIR / "edu_edges.json", encoding="utf-8"))

out = []

# --- Constraints ---
out.append("CREATE CONSTRAINT IF NOT EXISTS FOR (n:Node) REQUIRE n.id IS UNIQUE;\n")

# --- Nodes ---
for n in nodes:
    labels = f":{n['type']} :Node"
    props = f"id:'{n['id']}', label:'{n['label']}'"
    out.append(f"MERGE ({labels} {{{props}}});")

# --- Edges ---
for e in edges:
    rel = e["label"].upper()
    out.append(
        f"""
MATCH (a {{id:'{e['source']}'}}),(b {{id:'{e['target']}'}})
MERGE (a)-[:{rel}]->(b);
""".strip()
    )

cypher_file = EXPORT_DIR / "edu_graph.cypher"
cypher_file.write_text("\n".join(out), encoding="utf-8")

print(f"‚úÖ Neo4j Cypher generated at: {cypher_file}")

import json
from pathlib import Path
from rdflib import Graph, Namespace, RDF, RDFS, OWL, Literal

BASE_DIR = Path(__file__).resolve().parents[1]
GRAPH_DIR = BASE_DIR / "graph"
EXPORT_DIR = GRAPH_DIR / "exports"
EXPORT_DIR.mkdir(parents=True, exist_ok=True)

nodes = json.load(open(GRAPH_DIR / "edu_nodes.json", encoding="utf-8"))
edges = json.load(open(GRAPH_DIR / "edu_edges.json", encoding="utf-8"))

g = Graph()

EDU = Namespace("http://edupath.ai/ontology#")
ESCO = Namespace("http://data.europa.eu/esco/model#")

g.bind("edu", EDU)
g.bind("owl", OWL)
g.bind("rdfs", RDFS)
g.bind("esco", ESCO)

# --- Classes ---
types = set(n["type"] for n in nodes)
for t in types:
    g.add((EDU[t], RDF.type, OWL.Class))

# --- Individuals ---
for n in nodes:
    uri = EDU[n["id"]]
    g.add((uri, RDF.type, EDU[n["type"]]))
    g.add((uri, RDFS.label, Literal(n["label"])))

# --- Properties ---
for e in edges:
    prop = EDU[e["label"]]
    g.add((prop, RDF.type, OWL.ObjectProperty))
    g.add((EDU[e["source"]], prop, EDU[e["target"]]))

owl_file = EXPORT_DIR / "edu_ontology.owl"
g.serialize(destination=str(owl_file), format="xml")

print(f"‚úÖ OWL ontology generated at: {owl_file}")

üß† Core Principle (Very Important)

We DO NOT score nodes.
We score paths.

Because:

One job can be good in one region and bad in another

One program can lead to many jobs with different outcomes

Parents care about end outcomes, not abstract nodes

So the unit of ranking is:

Path = ordered sequence of nodes + edges

üü¶ Scoring Dimensions (Locked & Justified)

We will use 7 scoring components, each normalized to [0,1].

S1. Eligibility Score

Can the student realistically enter this path?

Inputs:

current stage (10th / 12th / UG)

stream (science/commerce/arts)

required exams

prerequisites

Eligible ‚Üí 1.0
Partially eligible ‚Üí 0.6
Not eligible ‚Üí 0.0

S2. Program‚ÄìJob Alignment

How directly does the program lead to the job?

Direct specialization ‚Üí job ‚Üí 1.0
Indirect but common ‚Üí 0.7
Weak/rare mapping ‚Üí 0.3


(derived from graph distance + edge types)

S3. Salary Score (Hybrid S3)

Uses your salary bands.

Formula (example):

salary_score = min(p75 / MAX_P75, 1.0)


Why p75?

p50 is ‚Äúaverage‚Äù

p75 reflects aspirational but realistic outcomes

S4. Demand Score (T3)

Uses forecast demand index.

demand_score = forecast_2026 / MAX_FORECAST


This future-proofs recommendations.

S5. Region Compatibility

How well does the path fit the user‚Äôs region preference?

Same city ‚Üí 1.0
Same state ‚Üí 0.8
Same country ‚Üí 0.6
Relocation needed ‚Üí 0.4


Critical for non-educated parents.

S6. Skill Gap Penalty

How many missing skills between student and job?

skill_gap_ratio = missing_skills / required_skills
skill_score = 1 - skill_gap_ratio


Certifications later reduce this penalty.

S7. Risk / Robustness Score

How resilient is this path?

Signals:

multiple employers?

multiple industries?

alternative programs?

Highly robust ‚Üí 1.0
Moderate ‚Üí 0.7
Fragile ‚Üí 0.4
